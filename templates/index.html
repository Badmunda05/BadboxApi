Python ਵਿੱਚ ਵਰਤੋ — same as catbox.moe
import requests

# File upload → direct URL
def upload(file_path):
    with open(file_path, "rb") as f:
        r = requests.post(
            "http://YOUR_IP:8000/api/upload",
            files={"file": f}
        )
    return r.text.strip()   # URL ਮਿਲਦੀ ਹੈ

url = upload("photo.jpg")
print(url)
# → http://YOUR_IP:8000/uploads/k9xm2pqr.jpg
